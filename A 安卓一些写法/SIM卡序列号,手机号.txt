    private String getPhoneNumber1() {
        TelephonyManager tm = (TelephonyManager) getSystemService(this.TELEPHONY_SERVICE);
        String phoneNumber1 = tm.getLine1Number();//SIM1卡手机号,有些是+86开头
        if (phoneNumber1.length() > 11) {
            phoneNumber1 = phoneNumber1.substring(phoneNumber1.length() - 11, phoneNumber1.length());
        }
        return phoneNumber1;
    }
