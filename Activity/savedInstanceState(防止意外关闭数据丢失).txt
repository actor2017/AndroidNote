package com.heima.mobilesafe_work.activity;
import android.os.Bundle;
import android.support.v7.app.AppCompatActivity;
import android.widget.TextView;
import com.heima.mobilesafe_work.R;
public class SettingActivity extends AppCompatActivity {
    
    private EditText et_test1;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_setting);
        /**
         * 刚进页面就判断有没有保存数据,如果有,就说明上次activity被
         * 意外关闭(来电话了,内存满了...),然后就读取数据填充到EditText
         * 中,让用户不用再第二次输入,增加用户体验
         */
        if (savedInstanceState != null) {
            String s11 = savedInstanceState.getString("s1", "");
            et_test1.setText(s11);
        }
        te_test1 = (EditText) findViewById(R.id.et_itemDescribe);
    }

    /**
     * 如果页面被意外关闭,就把TextView的值保存在数据捆outState中
     * 供下次打开的时候,直接填充的TextView中,让用户免于再输入
     * @param outState
     */
    @Override
    protected void onSaveInstanceState(Bundle outState) {
        String s1 = et_test1.getText().toString();
        outState.putString("s1",s1);
        super.onSaveInstanceState(outState);
    }
}
